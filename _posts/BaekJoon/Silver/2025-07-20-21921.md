```python
import sys

def input() :
    return sys.stdin.readline().strip()

n, x = map(int,input().split())
people = list(map(int,input().split()))
cnt = 1

if max(people) == 0 : print("SAD")
else :
    p = sum(people[0:x])
    v = p
    for i in range(x,n) :
        v -= people[i-x]
        v += people[i]
        if v > p:
            p = v
        elif v == p :
            cnt += 1
    print (p)
    print(cnt)
    ```
